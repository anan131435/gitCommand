git status 查看工作区的状态
git diff 查看工作区修改的内容
git add <fileName>
git commit -m’提交信息’
git pull
git push 提交到远程
git checkout — filename 撤销文件在工作区的修改
git reset HEAD file 撤销暂存区的修改至工作区
git rm  file移除文件
git log 查看提交日志
git reflog 查看命令历史
git remote add origin <git地址> 关联远程库
git clone <git 地址> 克隆远程库到本地
git reset —hard commitID 版本穿梭
git branch 
git branch <name>创建分支
git checkout <name>切换分支
git checkout -b <name>创建并切换分支
git merge <name> 合并name分支到当前分支
git branch -d <name> 删除分支
